
DROP TABLE IF EXISTS "public"."users" CASCADE;

CREATE TABLE users
(
	id SERIAL PRIMARY KEY,
    name text,
    email text
);


DROP TABLE IF EXISTS "public"."events";

CREATE TABLE events
(
	id SERIAL PRIMARY KEY,
    event_description text,
    event_date text,
    organizer_id int4 REFERENCES users(id),
    participant_id text
);
